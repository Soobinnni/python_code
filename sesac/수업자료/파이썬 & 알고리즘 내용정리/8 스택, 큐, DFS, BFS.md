```python
class Stack:
    def __init__(self,n):
        self.top = -1
        self.stack = [0]*n

    def push(self,data):
        if self.top == len(self.stack) - 1:
            return None
        self.top += 1
        self.stack[self.top] = data

    def pop(self):
        if self.top == -1:
            return None
        self.top -= 1
        return self.stack[self.top+1]

my_stack = Stack(10)
my_stack.push('alex')
print(my_stack)
print(my_stack.pop())
print(my_stack)
```
